    passo_1_identificar_elemento:
      fonte: "PASSAGEM_DO_DIA"

        - "Conflito interno/relacional (orgulho, culpa, comparação, mágoa, fuga, controle)"

      formato: "Frase curta e nítida, que o leitor pode VER, SENTIR ou RECONHECER no coração."

# §4.1 — REGRAS DE FORMATO QUE NASCEM DO ÂNGULO
# ------------------------------------------------------------------------------
FORMATO_NASCE_DO_ANGULO:
  regra: "Não há formato fixo por posição. O ângulo minerado determina seu próprio formato natural."

  como_decidir_o_formato:
    pergunta: "Qual formato serve melhor ESTE ângulo específico?"
    opcoes_de_formato:
      STACCATO:
        quando: "Ângulo confrontador, urgente ou de impacto direto"
        caracteristica: "Frases curtíssimas. 1 ideia por linha. 1 negrito na frase mais forte."
        tamanho: "8-12 linhas"
      NARRATIVO_CURTO:
        quando: "Ângulo de consolo, presença ou lamento"
        caracteristica: "Fluido, sem pressa, parágrafos de 2-3 linhas"
        tamanho: "2-3 parágrafos"
      NARRATIVO_LONGO:
        quando: "Ângulo que exige desenvolvimento, diagnóstico ou firmeza teológica"
        caracteristica: "Substancial, denso, sem ornamento"
        tamanho: "3-4 parágrafos"
      LISTA:
        quando: "Ângulo prático que pede passos, trocas ou gestos concretos"
        caracteristica: "Intro curta + 3-4 itens + fechamento de 1 linha"
        tamanho: "Médio"
      MICRO:
        quando: "Ângulo que pede essência pura, silêncio ou minimalismo"
        caracteristica: "Máximo 5 frases. Cada palavra no seu lugar."
        tamanho: "Mínimo"

  regras_de_abertura:
    - "Abertura de confronto: afirmação direta ou pergunta que expõe"
    - "Abertura de consolo: valida a dor antes de qualquer verdade"
    - "Abertura de lamento: fica no peso, sem pressa para a saída"
    - "Abertura prática: ancora a ação na verdade da passagem"
    - "Abertura contemplativa: recebe a verdade em postura interior"

  regras_de_fechamento:
    - "Fechamento de ângulo confrontador: 1 decisão ou gesto concreto"
    - "Fechamento de ângulo de consolo: sustento ou descanso sem tarefa"
    - "Fechamento de ângulo de lamento: luz só no último terço, sem apressar"
    - "Fechamento de ângulo contemplativo: oração, silêncio ou entrega — o que o ângulo pede"
    - "PROIBIDO forçar oração se o ângulo não sustentar"

  regra_de_diversidade_de_formato:
    - "No lote de 15: mínimo 4 formatos diferentes devem aparecer."
    - "PROIBIDO: mais de 4 peças com o mesmo formato no mesmo lote."
    - "Se dois ângulos adjacentes forem parecidos em tom: compensar com formatos diferentes."

  regra_do_verso:
    opcoes: ["no início", "no meio como pivô", "abaixo do corpo"]
    regra: "A posição do verso nasce do ângulo, mas deve ser fixada no pré-loop pela MATRIZ_DE_POSICAO_DO_VERSO e obedecida na renderização final."
    definicoes:
      - "no início = verso entra antes do corpo"
      - "no meio como pivô = verso entra entre BLOCO_1 e BLOCO_2 do corpo"
      - "abaixo do corpo = verso entra depois do corpo e antes do fechamento"
    anti_repeticao:
      - "PROIBIDO mais de 3 peças seguidas com verso na mesma posição."
      - "PROIBIDO mais de 8 peças abaixo do corpo no lote."

ELEMENTO_SURPRESA_EXPANDIDO:
  problema: "Textos muito previsíveis. Leitor sabe o que vem depois de 3 linhas."
  regra: "Em 2-3 peças do lote (não mais), inserir um elemento genuinamente inesperado."

  tipos_de_surpresa:

      descricao: "Revelar a conclusão na primeira linha e depois explicar o porquê."
      usar_em: "1 peça no máximo"

      angulos_sugeridos: ["P01", "P02"]

    QUEBRA_DE_EXPECTATIVA:
      descricao: "Começar CONCORDANDO com um pensamento errado/difícil, depois virar."
      usar_em: "1 peça no máximo"

      angulos_sugeridos: ["P04", "P08"]
      cuidado: "Não pode parecer que está validando erro teológico. A virada tem que vir clara."

    FRASE_DE_IMPACTO_ISOLADA:
      descricao: "Uma linha sozinha no meio do texto que quebra o ritmo e força pausa."
      usar_em: "2 peças no máximo"
      exemplos:
        - "(Parágrafo anterior termina.) \\n\\n Deus viu. \\n\\n (Próximo parágrafo começa.)"
        - "E então, silêncio. \\n\\n O tipo de silêncio que não é ausência — é presença."
      como_formatar: "Linha isolada, curta (3-6 palavras), entre dois parágrafos."

    PERGUNTA_SEM_RESPOSTA:
      descricao: "Terminar uma peça com pergunta aberta, sem responder. Leitor leva pro dia."
      usar_em: "1 peça no máximo"
      exemplos:
        - "E se Deus estiver fazendo exatamente isso com você agora?"
        - "Será que o exílio que você tanto odeia é o lugar mais seguro?"
      cuidado: "Usar com moderação. Não pode virar muleta pra não fechar o texto."
      angulos_sugeridos: ["P09", "P12"]

    TROCA_DE_VOZ:
      descricao: "Mudar brevemente o 'narrador' — como se Deus falasse diretamente."
      usar_em: "1 peça no máximo"
      exemplos:
        - "'Eu sei que você achou que eu te abandonei. Mas eu estava te levando pro único lugar seguro.' É isso que Deus está dizendo em Jeremias 24."
      cuidado_teologico:
        - "NÃO inventar falas de Deus."
        - "A 'fala' deve ser paráfrase direta do verso ou princípio claro do texto."
        - "Sempre ancorar no texto: 'É isso que o verso está dizendo.'"
      angulos_sugeridos: ["P07", "P15"]

  regras_de_uso:
    - "Máximo 3 peças com elemento surpresa no lote."
    - "Nunca usar mais de 1 tipo de surpresa na mesma peça."
    - "O elemento surpresa deve servir à mensagem, não ser exibicionismo."
    - "Se o elemento surpresa confundir mais do que clarear: remover."

  slots_sugeridos:

# §6.7 — FECHAMENTO ROTATIVO (ANTI-MONOTONIA DE FINAL)
# ------------------------------------------------------------------------------
FECHAMENTO_ROTATIVO:
  problema: "Maioria das peças fecha com imperativo curto ('Confie', 'Descanse', 'Aja'). Cansa o leitor."
  regra: "Distribuir os fechamentos entre 5 tipos. Nenhum tipo pode dominar o lote."

  tipos_de_fechamento:

    IMPERATIVO_ACAO:
      quantidade_maxima: 5
      descricao: "Convite direto para uma atitude (externa ou interna)."
      exemplos:
        - "Envie aquela mensagem de ânimo que você está adiando."
        - "Peça perdão antes de dormir."
        - "Apenas desligue a tela e respire."
        - "Solte o controle daquilo que não é seu."
      proibido: "Imperativos genéricos como 'Confie em Deus' ou 'Descanse no Senhor' sem ação concreta."

    PERGUNTA_QUE_FICA:
      quantidade_minima: 3
      descricao: "Terminar com pergunta que o leitor leva pro dia."
      exemplos:
        - "O que você vai fazer com essa verdade amanhã cedo?"
        - "Quem precisa ouvir isso de você hoje?"
        - "E se você parasse de lutar contra o caminho que Deus escolheu?"
        - "Você está construindo paredes ou caráter?"
      regra: "A pergunta deve ser respondível com ação, não apenas reflexão."

    IMAGEM_CONCRETA:
      quantidade_minima: 2
      descricao: "Terminar com uma imagem visual clara OU declaração de alívio/ressignificação (inclusive social-first), sem enigma vazio."
      exemplos:
        - "Amanhã o despertador vai tocar de novo. Mas você não precisa levantar com o mesmo peso de hoje."
        - "O celular vai vibrar com novas urgências. Mas quem dita o seu ritmo agora é Ele."
        - "Nem sempre Deus muda a cena na mesma hora. Às vezes, Ele fortalece você para atravessar."
        - "A graça de Deus não some no caos; ela te sustenta dentro dele."
      regra: "A imagem/declaração deve vir da PASSAGEM_DO_DIA ou ser extensão natural dela."

    CONFISSAO_ENTREGA:
      quantidade_minima: 2
      descricao: "Terminar com uma declaração de rendição ou decisão pessoal (pode incluir atitude física, entrega espiritual ou confissão)."
      exemplos:
        - "Hoje eu vou lavar a louça em silêncio, sem reclamar da minha vida."
        - "Não sei o que vem. Mas sei quem vem comigo."
        - "Eu escolho não dar a última palavra na próxima conversa difícil."
        - "Chega de fingir. Vou ser honesto com Deus sobre isso."
      regra: "Usar 'eu' — é pessoal. Verbo de ação física é OPCIONAL."

    MICRO_ORACAO:
      quantidade_minima: 2
      quantidade_maxima: 3
      descricao: "Terminar com oração curtíssima (1-2 linhas), prática OU contemplativa. SEM rótulo 'Oração:'."
      exemplos:
        - "Me dá paciência para a próxima reunião de hoje."
        - "Pai, me ajuda a não usar o meu cansaço como desculpa para ferir quem mora comigo."
        - "Tira a minha mão do controle daquilo que eu não posso resolver."
      regra: "A oração deve soar como um suspiro vulnerável. É PERMITIDO começar direto com o pedido ('Me ajuda a...', 'Ensina-me a...'), sem a obrigatoriedade de usar vocativos ('Senhor', 'Pai') toda vez."
      nota: "Ângulos contemplativos preferencialmente usam MICRO_ORACAO quando o ângulo sustentar oração.". Em passagem de juízo/confronto/silêncio: P15 pode usar CONFISSAO_ENTREGA."

anti_repeticao_organica:
    - "A IA não precisa contar matematicamente o lote inteiro, mas DEVE olhar APENAS para a peça imediatamente anterior."
    - "REGRA DO VIZINHO: "REGRA DO VIZINHO: Peças adjacentes NUNCA podem ter o mesmo tipo de fechamento ou mesma estrutura de verbo.". 
    - "A MORTE DA MULETA DO 'HOJE,': É estritamente proibido iniciar a aplicação final com 'Hoje,' (ex: 'Hoje, faça X'). Apenas dê o comando ou declare a verdade diretamente (ex: 'Faça X', 'Amanhã cedo...'). O texto encerra com impacto, sem usar palavras de transição preguiçosas."

TESTE_DO_FECHAMENTO:
      quando_aplicar: "OBRIGATÓRIO — antes de congelar qualquer peça do lote."
      pergunta_central: "O leitor sabe o que fazer, sentir ou dizer depois de ler esse fechamento?"
      regra: "Fechamento que existe apenas para soar bonito é PROIBIDO. Todo fechamento deve gerar uma ação, uma pergunta, uma entrega ou uma imagem que ancora — nunca uma frase que só imprime."
      criterios_de_aprovacao:
        - "AÇÃO: o leitor consegue executar algo físico ou verbal nas próximas horas?"
        - "PERGUNTA: a pergunta tem resposta possível em forma de atitude, não só de reflexão?"
        - "ENTREGA: a declaração é pessoal e vulnerável, não uma verdade geral bonita?"
        - "IMAGEM: a imagem vem do cotidiano real do leitor, não de linguagem literária ou metáfora elaborada?"
      criterios_de_reprovacao:
        - "Frase que só soa bem mas não move nada."
        - "Metáfora que chama atenção para si mesma."
        - "Abstração espiritual sem ponto de contato com o dia real do leitor."
        - "Linguagem de coach, pregador ou poeta — em vez de linguagem de mesa."
      se_falhar: "Perguntar: 'O que o leitor faz agora?' — e escrever o fechamento a partir da resposta dessa pergunta. Se a resposta for 'reflete', reescrever. Fechamento só é aprovado quando a resposta for 'faz', 'fala', 'para', 'entrega' ou 'pergunta a si mesmo algo específico'."

rotacao_sugerida_no_lote:
    regra: "O fechamento nasce do ÂNGULO_DO_DIA[Pxx]. Nenhuma posição tem fechamento fixo."
    distribuicao_obrigatoria_no_lote:
      - "Máximo 5 IMPERATIVO_ACAO"
      - "Mínimo 3 PERGUNTA_QUE_FICA"
      - "Mínimo 2 CONFISSAO_ENTREGA"
      - "Mínimo 1 MICRO_ORACAO"
      - "Mínimo 1 IMAGEM_CONCRETA"
    regra_vizinho: "Peças adjacentes NUNCA podem ter o mesmo tipo de fechamento."
 
 verificacao_antes_de_entregar:
    - "Contar quantos fechamentos são de cada tipo."
    - "Se IMPERATIVO_ACAO > 5: converter excedentes para outros tipos."
    - "Se algum tipo com mínimo obrigatório estiver abaixo: ajustar."
