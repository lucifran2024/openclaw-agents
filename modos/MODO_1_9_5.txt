MOTOR_ABERTURA_DIVERSIFICADA:
  problema: "Muitas peças começam explicando o texto bíblico ou com abstração genérica."
  regra: "Distribuir OBRIGATORIAMENTE os 15 textos entre 5 tipos de abertura."

    passo_2: "Distribuir os 15 perfis entre os 5 tipos ANTES de começar."
    passo_3: "Se ao final o lote não cumprir os mínimos, REESCREVER as aberturas."

  distribuicao_organica_por_clima:
    regra_soberana: "A PASSAGEM_DO_DIA dita a abertura, não uma tabela fixa. A IA deve distribuir os 5 tipos de abertura organicamente ao longo dos 15 perfis, garantindo que nenhum tipo se repita mais de 4 vezes no lote."
    guia_de_intuicao:
      - "Se o perfil pede confronto (ex: P08), prefira PERGUNTA_CORTANTE."
      - "Se o perfil pede consolo (ex: P07), prefira CONFISSAO_IDENTIFICACAO ou CENA_COTIDIANO."
      - "Se o perfil pede ação (ex: P05), prefira AFIRMACAO_DE_CHOQUE."
    nota: "A fluidez e a coerência com o texto bíblico valem mais do que gabaritar cotas."
  regra: "Distribuir organicamente os 15 textos entre 5 tipos de abertura, obedecendo ao clima da PASSAGEM_DO_DIA."

  tipos_de_abertura:

    CENA_COTIDIANO:
      quantidade_minima: 1
      quantidade_maxima: 3
      descricao: "Começar com situação real do dia a dia (cena curta OU diagnóstico direto). Sem drama exagerado, sem roteirização cinematográfica e sem cenografia corporal decorativa."
      proibido: "Abertura imersiva/filmada ('Você senta...', 'A porta bate...', 'O relógio marca...') ou somática ('ar bate no peito', 'nó na garganta', 'peito acelerado') quando isso virar decoração e não conflito."
      regra_de_validade: "A PRIMEIRA frase deve trazer conflito, decisão, culpa, medo, pressão ou tensão relacional. Se a frase só ambienta (silêncio/casa/noite/ar/corpo), INVALIDAR e reescrever."
      exemplos:
        - "Seu dia começa e a cabeça já está negociando com um problema."
        - "A lista de demandas cresce mais rápido do que a sua energia."
        - "Você perdeu a paciência cedo demais hoje, e isso te incomodou."
        - "Você percebeu hoje que está reagindo no automático e ferindo gente."
      perfis_preferidos: ["P07", "P10"]
      gatilho: "Usar quando PASSAGEM falar de cansaço, espera, dor, luto, solidão."

    PERGUNTA_CORTANTE:
      quantidade_minima: 3
      quantidade_maxima: 4
      descricao: "Começar com pergunta que incomoda, sem resposta imediata."

        - "O que você faz quando Deus demora?"
        - "Até quando você vai fingir que está tudo bem?"
        - "E se o problema não for a circunstância, mas o coração?"
      perfis_preferidos: ["P05", "P08", "P09", "P14"]
      gatilho: "Usar quando PASSAGEM trouxer confronto, escolha, decisão."

      quantidade_minima: 2
      quantidade_maxima: 3
      descricao: "Começar admitindo algo humano. Usar 'eu' ou 'a gente' (1x só)."

  alternativas_de_virada:

    CONTRASTE_SILENCIOSO:
      descricao: "Não usar conjunção adversativa. Deixar o contraste implícito pela justaposição."
      exemplos:
        - "O rei construía palácios. Deus media o caráter."
        - "Eles esperavam vitória. Deus enviou exílio — e era isso que ia salvar."
        - "O cenário gritava derrota. A promessa sussurrava outra coisa."

    VIRADA_POR_PERGUNTA:
      descricao: "Usar pergunta como pivô que força reflexão."
      exemplos:
        - "E se o exílio não for castigo, mas proteção?"
        - "Será que o problema é o caminho ou quem você esperava encontrar nele?"
        - "O que muda se você olhar isso pela lente de quem já viu o fim?"
      usar_em: "mínimo 2 peças"

    VIRADA_PELO_VERSO:
      descricao: "Deixar o próprio verso bíblico fazer a virada, sem preparação."
      exemplos:
        - "Aí Deus fala: 'Eu mesmo reunirei o remanescente.' Ponto."
        - "E então vem o verso: 'Darei a eles um coração para me conhecerem.' Isso muda tudo."
      usar_em: "mínimo 2 peças"

    VIRADA_POR_ZOOM:
      descricao: "Sair do macro (história bíblica) pro micro (vida do leitor) ou vice-versa."
      exemplos:
        - "Isso parece história antiga. Até você lembrar da última reunião de trabalho."
        - "Dois cestos de figos. Parece distante. Mas você também está sendo avaliado hoje."
        - "Jeremias falou pro rei. E pra você, o que Deus está falando?"
      usar_em: "mínimo 2 peças"

    VIRADA_POR_CONFISSAO:
      descricao: "Admitir dificuldade antes de apontar a saída. Cria empatia."
      exemplos:
        - "Eu sei, é difícil acreditar nisso quando a conta não fecha."
        - "Parece bonito no papel. Na vida real, dói."
        - "Não vou mentir: essa parte da Bíblia me incomoda também."
      usar_em: "mínimo 2 peças"

  distribuicao_obrigatoria:
    "Mas Deus/Mas o Senhor/Mas a promessa": "máximo 3 peças"
    "Contraste silencioso": "mínimo 3 peças"
    "Virada por pergunta": "mínimo 2 peças"
